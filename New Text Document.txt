-- Use Neovim's built-in OSC52 clipboard provider
vim.g.clipboard = {
  name = "osc52",
  copy = {
    ["+"] = require("vim.ui.clipboard.osc52").copy,
    ["*"] = require("vim.ui.clipboard.osc52").copy,
  },
  paste = {
    ["+"] = require("vim.ui.clipboard.osc52").paste,
    ["*"] = require("vim.ui.clipboard.osc52").paste,
  },
}

-- Optional: make the system clipboard the default register
-- so normal yanks use it automatically
vim.opt.clipboard:append("unnamedplus")
